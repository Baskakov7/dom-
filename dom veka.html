    function getSelectedPricePerM2() {

      const opt = qs('#type').selectedOptions[0];

    function getMultiplier(selId) {

      const opt = qs('#' + selId).selectedOptions[0];

    }

    function calculatePrice() {

      const area = Number(qs('#area').value || 0);

      const pricePerM2 = getSelectedPricePerM2();

      const finishMult = getMultiplier('finish');

      const energyMult = getMultiplier('energy');

      if (!area || !pricePerM2) {

        qs('#price').textContent = "Стоимость: —";

        qs('#usePriceForMortgage').disabled = true;

        return;

      }

      const base = area * pricePerM2;

      const total = base * finishMult * energyMult;

      const min = total * 0.95;

      const max = total * 1.1;

      qs('#price').textContent = `Стоимость: ${fmtCurrency(min)} — ${fmtCurrency(max)}`;

      qs('#usePriceForMortgage').disabled = false;

      qs('#usePriceForMortgage').dataset.value = Math.round(total);

    }

    window.calculatePrice = calculatePrice;

    // ========== ИПОТЕЧНЫЙ КАЛЬКУЛЯТОР ==========

    function calcMortgagePayment(principal, annualRate, years) {

      const months = years * 12;

      const r = (annualRate / 100) / 12;

      if (r === 0) return principal / months;

      return principal * (r * Math.pow(1 + r, months)) / (Math.pow(1 + r, months) - 1);

    }

    qs('#calcMortgage').addEventListener('click', () => {

      const price = Number(qs('#homePrice').value || 0);

      const down = Number(qs('#downPayment').value || 0);

      const rate = Number(qs('#rate').value || 0);

      const years = Number(qs('#years').value || 0);


      if (price <= 0 || rate <= 0 || years <= 0 || down < 0 || down >= price) {

        qs('#mortgageResult').textContent = "Проверьте исходные данные";

        return;

      }

      const principal = price - down;

      const pay = calcMortgagePayment(principal, rate, years);

      const incomeNeeded = pay / 0.35; // условно 35% дохода

      qs('#mortgageResult').innerHTML =

        `Ежемесячный платёж: <strong>${fmtCurrency(pay)}</strong><br>` +

        `Остаток по кредиту: ${fmtCurrency(principal)} • Требуемый доход: ~${fmtCurrency(incomeNeeded)}`;

    });
